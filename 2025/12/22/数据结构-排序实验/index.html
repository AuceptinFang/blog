题目

内部排序算法比较

对7种常用的内部排序算法进行比较：直接插入排序、希尔排序、冒泡排序、快速排序、简单选择排序、堆排序、二路归并排序。
待排序表的表长不少于100；其中的数据要用伪随机数产生程序产生；至少要用5组不同的输入数据作比较；比较的指标为关键字参加的比较次数和数据元素的移动次数（关键字交换计为3次移动）。
对不同的输入表长做试验，观察两个指标相对于表长的变化关系，也可以对稳定性做验证。

统计成绩

给出n个学生的m门考试的成绩表，每个学生的信息由学号、姓名以及各科成绩组成。对学生的考试成绩进行有关统计，并打印统计表。
①按总分由高到低的顺序，打印出名次表，分数相同的为同一名次；
②按名次打印出每个学生的学号、姓名、总分以及各科成绩。

一

一些经验，rust对于类型要求很严格，所以用作索引的一律用usize，因为for循环里的i默认是，统计数据用u64，如果u32溢出程序会直接panic

use std::mem::swap;
use tracing::info;
use tracing_subscriber::filter::LevelFilter;

pub struct  Benchmark{
    compare_time : u64,
    move_time : u64,
}

struct Sort{
    bench : Benchmark,
}

impl Sort{

    pub fn new() -> Sort{
        Sort{
            bench : Benchmark{
                compare_time : 0,
                move_time : 0,
            }
        }
    }

    pub fn compare(&mut self, num1 : i32, num2 : i32) -> bool{
        self.bench.compare_time += 1;
        num1 > num2
    }

    pub fn insert_sort(&mut self, nums : &mut Vec<i32>){
        let mut temp = 0;
        for i in 1..nums.len(){
            if self.compare(nums[i-1], nums[i]) { //如果小于，直接跳过
                temp = nums[i];
                let mut j = i;

                while j > 0 && !self.compare(temp, nums[j-1]){ // 与第j-1个元素相比较
                    nums[j] = nums[j-1];
                    self.bench.move_time += 1;
                    j -= 1;
                }
                nums[j] = temp;
                self.bench.move_time +=1;
            }
        }
    }

    pub fn shell_sort(&mut self, nums : &mut Vec<i32>) {
        let dlta = vec![8,4,2,1];
        for k in 0..dlta.len(){
            self.shell_insert(nums, dlta[k]);
        }
    }

    fn shell_insert(&mut self, nums : &mut Vec<i32>, d : u32){
        // d为步长
        let dd = d as usize;
        // 分成d次插入排序
        for i in dd..nums.len(){
            if self.compare(nums[i - dd],nums[i]){
                let temp = nums[i];
                let mut j = i;

                while j >= dd && self.compare(nums[j - dd], temp) {
                    nums[j] = nums[j - dd];
                    self.bench.move_time += 1;
                    j -= dd;
                }

                nums[j] = temp;
                self.bench.move_time += 1;
            }
        }
    }

    pub fn bubble_sort(&mut self, nums : &mut Vec<i32>){
        let mut change = true;
        let n = nums.len();
        for i in 0..n{
            if change == false{
                return;
            }
            change = false;
            for j in 1..n - i{
                if self.compare(nums[j-1], nums[j]){
                    // 交换，记录
                    nums.swap(j-1, j);
                    self.bench.move_time += 3;
                    change = true;
                }
            }
        }
    }

    pub fn quick_sort(&mut self, nums : &mut Vec<i32>, low: usize, high: usize){
        if high > low {
            let pi = self.partition(nums, low, high);

            if pi > low { // 避免pi = 0发生溢出的情况
                self.quick_sort(nums, low, pi - 1);
            }

            self.quick_sort(nums, pi + 1, high);
        }
    }

    fn partition(&mut self, nums : &mut Vec<i32>, low: usize, high: usize) -> usize{
        let pivot = nums[high as usize];
        let mut index : usize = low as usize;
        for j in low..high {
            if self.compare(pivot, nums[j]){
                index += 1;
                nums.swap(index -1,j);
                self.bench.move_time += 3;

            }
        }
        nums.swap(index,high as usize);
        index
    }

    pub fn select_sort(&mut self, nums : &mut Vec<i32>){
        let t = nums.len();
        for i in 0..t-1{
            let target = t - 1 - i;
            let k = self.find_max(nums, t-i);
            if k != target{
                nums.swap(k,target);
                self.bench.move_time += 3;
            }
        }
    }

    fn find_max(&mut self, nums : &mut Vec<i32>, len:usize) -> usize{
        let mut max = nums[0];
        let mut max_index = 0;
        for i in 1..len{
            if self.compare(nums[i],max){
                max_index = i;
                max = nums[i];
            }
        }
        max_index
    }

    pub fn heap_sort(&mut self, nums : &mut Vec<i32>){
        let len = nums.len();
        let last_parent = (len / 2) - 1;

        for i in (0..=last_parent).rev(){
            self.sift_down(nums, i, len - 1);
        }

        for i in (1..=len - 1).rev(){
            nums.swap(0, i);
            self.bench.move_time += 3;

            self.sift_down(nums, 0, i-1);

        }
    }

    fn sift_down(&mut self, nums : &mut Vec<i32>, start : usize, end : usize){
        let mut parent = start;
        let mut child = parent * 2 + 1;
        while child <= end{
            if child + 1 <= end && self.compare(nums[child + 1], nums[child]){
                child += 1;
            }
            if self.compare(nums[parent],nums[child] ) {
                return;
            }else{
                nums.swap(child, parent);
                self.bench.move_time += 3;
                parent = child;
                child = parent * 2 + 1;
            }
        }
    }

    pub fn merge_sort(&mut self, nums : &mut Vec<i32>, left : usize, right : usize){
        if left < right{
            let mid = left + (right - left) / 2;
            self.merge_sort(nums, left, mid);
            self.merge_sort(nums, mid + 1, right);
            self.merge(nums, left,mid, right);
        }
    }

    fn merge(&mut self, nums : &mut Vec<i32>, left : usize,mid : usize, right : usize){
        let l1 = mid - left + 1;
        let l2 = right - mid;

        let v1 = nums[left..left + l1].to_vec();
        let v2 = nums[mid + 1..mid + 1 + l2].to_vec();
        self.bench.move_time += (l1 + l2) as u64;

        let mut i = 0;
        let mut j = 0;
        let mut k = left;

        while i<l1 && j<l2{
            if self.compare(v1[i], v2[j]){
                nums[k] = v2[j];
                self.bench.move_time += 1;
                j += 1;
            }else {
                nums[k] = v1[i];
                self.bench.move_time += 1;
                i += 1;
            }
            k += 1;
        }

        while i < l1{
            nums[k] = v1[i];
            self.bench.move_time += 1;
            i += 1;
            k += 1;
        }

        while i < l2{
            nums[k] = v2[j];
            self.bench.move_time += 1;
            j += 1;
            k += 1;
        }

    }
}

实现

一道排序算法模板题，但是没有标准库可用，同时要求输出时附带名次且同分同名

既然测试的结果快排最好，那接下来就在快排的基础上改，首先把它变成一个泛型函数：

pub fn quick_sort <T>(&mut self, nums : &mut [T], low: usize, high: usize)
where
    T : PartialOrd + Clone,
{
    if high > low {
        let pi = self.partition(nums, low, high);

        if pi > low { // 避免pi = 0发生溢出的情况
            self.quick_sort(nums, low, pi - 1);
        }

        self.quick_sort(nums, pi + 1, high);
    }
}

fn partition <T>(&mut self, nums : &mut [T], low: usize, high: usize) -> usize
where
    T : PartialOrd + Clone,
{
    let pivot = nums[high].clone();
    let mut index = low ;
    for j in low..high {
        if self.compare(pivot.clone(), nums[j].clone()){
            index += 1;
            nums.swap(index -1,j);
            self.bench.move_time += 3;

        }
    }
    nums.swap(index,high);
    index
}

pub fn compare<T: PartialOrd>(&mut self, num1 : T, num2 : T) -> bool{
    self.bench.compare_time += 1;
    num1 > num2
}

compare函数的比较对象并没有写成引用，这导致每次调用compare会移交所有权，而循环在继续，所以不得不.clone()，但这又会造成不必要的开销。i32类型时没有所有权问题是因为i32具有copy特征，其表示按位复制，当实现了copy特征后传参就会变成按位复制而非移动所有权。如果我传入一个指针，它会在函数结束被free掉，在自己的生命周期结束再被free，显然rust不会允许这种事情发生。涉及堆的、文件句柄、连接等的数据都无法实现copy特征

按理说compare的参数应当是引用，这是最标准的写法，这里为避免大面积修改clone了事了

接下来是Student结构体的实现，单独划分出一个字段记录名次，同时实现必要的特征

    pub fn new(id : u64, name : String, math : i32, chinese : i32, english : i32) -> Self{
        Self{
            id,
            name,
            math,
            chinese,
            english,
            order : 0,
        }
    }
}

impl fmt::Display for Student{
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        let total = self.math + self.chinese + self.english;
        write!(f,"学号：{}, 姓名：{} 总分:{}, 语文：{} 数学：{}英语：{}",self.id,self.name,total,self.chinese,self.math,self.english)
    }
}

impl PartialEq for Student{
    fn eq(&self, other: &Self) -> bool{
        let t1 = self.math + self.chinese + self.english;
        let t2 = other.math + other.chinese + other.english;
        t1 == t2
    }
}

impl PartialOrd for Student{
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        let t1 = self.math + self.chinese + self.english;
        let t2 = other.math + other.chinese + other.english;

        t1.partial_cmp(&t2)
    }
}

pub fn sort(mut s : Vec<Student>){
    let mut sort_instance = Sort::new();
    let len = s.len();
    sort_instance.quick_sort(&mut s, 0, len - 1);
    let mut current_score = 0;
    let mut current_order = 0;
    for i in (0..len).rev(){
        let order = len - 1 - i;
        let score = s[i].chinese + s[i].math + s[i].english;

        let mut index = order;
        if current_score == score {
            index = current_order;
        }else{
            current_order = index;
            current_score = score;
        }
        info!("第{}名，{}",index + 1,s[i]);
    }
}

fn partition_hoare <T>(&mut self, nums: &mut [T], low: usize, high: usize) -> usize
where 
    T: PartialOrd,
    {
        let pivot_index = low;
        let mut l = low + 1;
        let mut h = high;

        loop{
            while l <= h && self.compare(&nums[h],&nums[pivot_index] ){
                h -= 1;
            }
            
            while l <= h && !self.compare(&nums[l],&nums[pivot_index] ){
                l += 1;
            }

            if l > h {
                break;
            }
            nums.swap(h,l);
            self.bench.move_time += 3;
            l += 1;
            h -= 1;
        }
        nums.swap(pivot_index,h);
        self.bench.move_time += 3;
        h
    }

测试结果是几乎没有变化
